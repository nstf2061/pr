#include <stdio.h>

int countNumbers(int r) {
    if (r == 1) return 2; // 5 або 9
    if (r == 2) return 4; // 55, 59, 95, 99

    long long dp[r+1][2][2]; // dp[length][last_digit][second_last_digit]
    
    dp[2][0][0] = 1; // 55
    dp[2][0][1] = 1; // 59
    dp[2][1][0] = 1; // 95
    dp[2][1][1] = 1; // 99

    for (int i = 3; i <= r; i++) {
        dp[i][0][0] = dp[i-1][1][0]; // ...95 -> ...555
        dp[i][0][1] = dp[i-1][0][0] + dp[i-1][1][0]; // ...55 -> ...5559 або ...955 -> ...9559
        dp[i][1][0] = dp[i-1][0][1] + dp[i-1][1][1]; // ...59 -> ...595 або ...99 -> ...995
        dp[i][1][1] = dp[i-1][0][0] + dp[i-1][1][0]; // ...55 -> ...599 або ...95 -> ...999
    }

    long long result = dp[r][0][0] + dp[r][0][1] + dp[r][1][0] + dp[r][1][1];
    
    return result;
}

int main() {
    int r;
    printf("Введіть значення р: ");
    scanf("%d", &r);

    if (r < 1 || r > 30) {
        printf("Значення р має бути в межах від 1 до 30.\n");
        return 1;
    }

    int result = countNumbers(r);
    printf("Кількість чисел з %d розрядів: %d\n", r, result);

    return 0;
}
