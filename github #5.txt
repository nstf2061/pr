#include <stdio.h>

#define MOD 12345

int countSequences(int n) {
    if (n == 1) return 2;
    if (n == 2) return 4;

    int dp0[n+1], dp1[n+1], dp2[n+1];

    dp0[1] = 1; dp1[1] = 1; dp2[1] = 0;
    dp0[2] = 2; dp1[2] = 1; dp2[2] = 1;

    for (int i = 3; i <= n; i++) {
        dp0[i] = (dp0[i-1] + dp1[i-1] + dp2[i-1]) % MOD;
        dp1[i] = dp0[i-1] % MOD;
        dp2[i] = dp1[i-1] % MOD;
    }

    int result = (dp0[n] + dp1[n] + dp2[n]) % MOD;
    return result;
}

int main() {
    int n;
    printf("Введіть довжину послідовності n: ");
    scanf("%d", &n);

    if (n <= 1 || n >= 10000) {
        printf("Значення n має бути в межах від 2 до 9999.\n");
        return 1;
    }

    int result = countSequences(n);
    printf("Кількість шуканих послідовностей: %d\n", result);

    return 0;
}
