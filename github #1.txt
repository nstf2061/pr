#include <stdio.h>

double calculate_time(int t1, int t2, int t3) {
        double part1 = 1.0 / t1;
    double part2 = 1.0 / t2;
    double part3 = 1.0 / t3;
    
    
    double total_rate = part1 + part2 + part3;
    
        double total_time = 1.0 / total_rate;
    
    
    return total_time;
}

int main() {
    int t1, t2, t3;
    
        printf("Введіть три числа t1, t2, t3: ");
    scanf("%d %d %d", &t1, &t2, &t3);
    
       double result = calculate_time(t1, t2, t3);
    
        printf("Час, за який гості з'їдять пиріг: %.2f годин\n", result);
    
    return 0;
}
