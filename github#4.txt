#include <stdio.h>

int main() {
    int n;
    printf("Введіть натуральне число n: ");
    scanf("%d", &n);

    if (n <= 1 || n >= 150) {
        printf("Число має бути в межах від 2 до 149.\n");
        return 1;
    }

    int count = 0;

    for (int m = 1; m < n; m++) {
        int quotient = n / m;
        int remainder = n % m;
        
        // Перевірка умови рівного дільника
        if (quotient == remainder) {
            count++;
        }
    }

    printf("Кількість рівних дільників: %d\n", count);

    return 0;
}
