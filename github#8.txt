#include <stdio.h>
#include <string.h>

unsigned long long factorial(int n) {
    if (n == 0 || n == 1) return 1;
    unsigned long long result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

unsigned long long countAnagrams(char* word) {
    int length = strlen(word);
    int count[256] = {0}; // Масив для підрахунку частот кожної букви

    for (int i = 0; i < length; i++) {
        count[(int)word[i]]++;
    }

    unsigned long long result = factorial(length);

    for (int i = 0; i < 256; i++) {
        if (count[i] > 1) {
            result /= factorial(count[i]);
        }
    }

    return result;
}

int main() {
    char word[15]; 
    printf("Введіть слово: ");
    scanf("%s", word);

    unsigned long long anagrams = countAnagrams(word);
    printf("Кількість можливих анаграм: %llu\n", anagrams);

    return 0;
}
